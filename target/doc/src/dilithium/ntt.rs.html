/// Forward NTT, in-place. No modular reduction is performed after additions or subtractions. Output vector is in bitreversed order.
///
/// # Arguments
///
/// * 'a' - input/output coefficient array of size N
pub fn ntt(a: &mut [i32]) {
    let mut k: usize = 0;
    let mut len: usize = 128;

    while len > 0 {
        let mut start: usize = 0;
        while start < N {
            k += 1;
            let zeta = ZETAS[k] as i64;
            let mut j = start;
            while j < (start + len) {
                let t = crate::reduce::montgomery_reduce(zeta.wrapping_mul(a[j + len] as i64));
                a[j + len] = a[j] - t;
                a[j] += t;
                j += 1;
            }
            start = j + len;
        }
    len >>= 1;
    }
}

/// Inverse NTT and multiplication by Montgomery factor 2^32. In-place. No modular reductions after additions or subtractions; input coefficients need to be smaller than Q in absolute value. Output coefficient are smaller than Q in absolute value.
/// 
/// # Arguments
/// 
/// * 'a' - input/output coefficient array of size N
pub fn invntt_tomont(a: &mut [i32]) {
    let mut k: usize = 256;
    let mut len: usize = 1;
    const F: i64 = 41978; // mont^2/256

    while len < N {
        let mut start: usize = 0;
        while start < 256 {
            k -= 1;
            let zeta = -ZETAS[k] as i64;
            let mut j = start;
            while j < (start + len) {
                let t = a[j];
                a[j] = t + a[j + len];
                a[j + len] = t - a[j + len];
                a[j + len] = crate::reduce::montgomery_reduce(zeta.wrapping_mul(a[j + len] as i64));
                j += 1
            }
            start = j + len;
        }
        len <<= 1;
    }
    for j in 0..N {
        a[j] = crate::reduce::montgomery_reduce(F.wrapping_mul(a[j] as i64));
    }
}
